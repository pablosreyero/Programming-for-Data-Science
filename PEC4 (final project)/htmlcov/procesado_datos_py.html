9def df_nueva_variable(df_eje1): 

10 """ 

11 Esta función se encarga de añadir una variable más al dataframe creado en 

12 el ejercicio 1.2. La función se ha diseñado de tal forma, que la variable 

13 refleje el número de días que la serie ha estado en emisión. 

14 

15 Parámetros de entrada: 

16 - df_eje1 (pandas df) -- esta variable se corresponde con el dataframe del 

17 ejercicio 1.2. 

18 

19 Parámetros de salida: 

20 - df_eje1 (pandas df) -- el nuevo dataframe con la variable añadida. 

21 - df_final (pandas df) -- el dataframe ordenado de manera descendente (de  

22 mayor a menor número de días de serie/doc/pelicula emitida) 

23 """ 

24 

25 # creamos una lista donde guardaremos los dias 

26 lista_dias = [] 

27 for _, row in df_eje1.iterrows(): 

28 if pd.isna(row['first_air_date']) or pd.isna(row['last_air_date']): 

29 lista_dias.append("-300000") 

30 else: 

31 a = row['first_air_date'] 

32 b = row['last_air_date'] 

33 fecha1 = datetime.strptime(a, "%Y-%m-%d") 

34 fecha2 = datetime.strptime(b, "%Y-%m-%d") 

35 dias_emision = fecha2 - fecha1 

36 dias_emision = dias_emision.days # esto es un "int" 

37 if dias_emision < 0: 

38 row['first_air_date'] = b 

39 row['last_air_date'] = a 

40 lista_dias.append(abs(dias_emision)) 

41 else: 

42 lista_dias.append(dias_emision) 

43 

44 # ahora convertimos la lista de strings a una lista de ints, para poder 

45 # ordenar el dataframe y mostrar los 10 registros con más días emitidos 

46 lista_dias = list(map(int, lista_dias)) 

47 

48 # añadimos una columna vacía en el dataframe con el nombre: "air_days" 

49 # la columna la vamos a añadir despues de la columna: "last_air_date" 

50 # para ello necesitamos el índice de la columna "last_air_date" + 1. 

51 idx_last_air_date = df_eje1.columns.get_loc("last_air_date") 

52 df_eje1.insert(idx_last_air_date+1, 'air_days', lista_dias) 

53 

54 # ahora ordenamos el dataframe según el número de días en emisión 

55 df_final = df_eje1.sort_values(by = 'air_days', ascending = False) 

56 return df_eje1, df_final
